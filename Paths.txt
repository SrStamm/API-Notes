USERS

GET		"/"			Obtiene una lista con todos los usuarios
GET		"/me"			Obtiene el usuario logeado
GET		"/{id}"			Obtiene el usuario de id especificado
GET		"/{id}/admin"		Obtiene datos sensibles de id especificado
POST		"/"			Crea un nuevo usuario
PATCH		"/me"			Actualiza los datos especificados del usuario actual
PUT		"/{id}"			Actualiza un usuario completo de id especificado
DELETE		"/me"			Elimina el usuario actual
DELETE		"/{id}"			Elimina el usuario de id especificado
